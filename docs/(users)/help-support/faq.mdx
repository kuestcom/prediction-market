title: FAQ

---

import { Accordions, Accordion } from 'fumadocs-ui/components/accordion'

<Accordions type="single" collapsible>
  <Accordion title="What fee do I pay to trade?" value="fees">
    Market orders use the live configured taker fee: <strong><TradingFeeDisplay /></strong>.
    If your order has a referral, the operator fee is split between affiliate and operator.
  </Accordion>

  <Accordion title="Where do I cancel open orders?" value="cancel-orders">
    Open <strong>/portfolio</strong>, switch to <strong>Open Orders</strong>, then cancel single orders
    or use <strong>Cancel all</strong> when available.
  </Accordion>

  <Accordion title="How do I claim winnings?" value="claim">
    Claim from the Portfolio winnings card or from the resolved market panel using
    <strong>Claim winnings</strong> when displayed.
  </Accordion>

  <Accordion title="How do I claim affiliate fees?" value="affiliate-claim">
    Open <strong>/settings/affiliate</strong> and use the <strong>Claim</strong> button on the
    <strong>Affiliate Commissions</strong> card once the balance is available for your deposit wallet.
  </Accordion>

  <Accordion title="What if a market is disputed?" value="dispute">
    Watch the market timeline status (Dispute window/Disputed/Final review).
    When enabled, the market shows a dispute action linking to the oracle dispute flow.
  </Accordion>

  <Accordion title="Where is transaction history export?" value="csv">
    Go to <strong>/portfolio</strong> → <strong>History</strong> and use the export icon
    to download CSV for the current filtered view.
  </Accordion>
</Accordions>
